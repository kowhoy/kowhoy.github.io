Scala循环

  1. for循环
  • 使用 yield生成新数组
val a = Array("apple", "banana", "orange")

for (e <- a) println(e)

for (e <- a) {
    val s = e.toUpperCase
    println(s)
}

//生成新的数组
val newArray = for (e <- a) yield e.toUpperCase

val newArray = for (e <- a) yield {
    val s = s.toUpperCase
    s
}

for循环中的多次计数器

for {
    i <- 1 to 2
    j <- 1 to 2
} println(s"i=$i, j=$j")

val array = Array.ofDim[Int](2, 2)

scala> array(0)(0) = 0

scala> array(0)(1) = 1

scala> array(1)(0) = 2

scala> array(1)(1) = 3

scala> array
res18: Array[Array[Int]] = Array(Array(0, 1), Array(2, 3))

scala> for {
     | i <- 0 to 1
     | j <- 0 to 1
     | } println(s"($i)($j) = ${array(i)(j)}")
(0)(0) = 0
(0)(1) = 1
(1)(0) = 2
(1)(1) = 3

在for语句中使用if(卫语句)

for {
    i <- 1 to 10
    if i%2 == 0
} println(i)

for {
    i <- 1 to 10
    if i > 3
    if i < 5
} println(i)

for {
    i <- 1 to 10
    if (i > 3 && i < 5)
} println(i)

不带卫语句的循环可以使用 map

var fruits = scala.collection.mutable.ArrayBuffer[String]()
fruits += "apple"
fruits += "banana"
fruits += "orange"

val out = for (e <- fruits) yield e.toUpperCase

val out = fruits.map(_.toUpperCase)

val in = out.map(_.toLowerCase)

scala 中的break 和 continue

在scala中是没有break和 continue关键字的

import util.control.Breaks._ 引入，breakable 在循环体内为continue， 在循环体外为break

import util.control.Breaks._

object BreakAndContinurDemo extends App {
    println("=== Break example ===")
    breakable {
        for (i <- 1 to 10) {
            println(i)
            if (i > 4) break
        }
    }

    println("=== Continue example ===")
    val searchStr = "peter piper picked a peck of pickled peppers"
    var numPs = 0

    for (i <- 0 until searchStr.length) {
        breakable {
            if (searchStr.charAt(i) != 'p') {
                break
            } else {
                numPs += 1
            }
        }
    }

    println("Found " + numPs + " p's in the string")
}
