用python将scala_cookbook保存为md

The Scala Cookbook在线阅读

由于布局和广告，想将每章节保存到本地，方便阅读

主要分为两个部分:

  1. 保存文章的html(save_art_html.py)

  2. 将文章部分截取并保存为md(trans_2_md.py)

两个部分可以合并为一

目录结构：

├── result

├── save_art_html.py

├── source

└── trans_2_md.py

curl http://scalacookbook.com/ > ./source/index.html

save_art_html.py

from lxml import etree
import os

with open("./source/index.html", "r") as f:
    index_html = f.read()

tree = etree.HTML(index_html)

urls_content = tree.xpath("//a")

urls_list = [i.attrib['href'] for i in urls_content]

for url in urls_list:
    url_list = url.split("/")
    url_name = url_list[-1]
    os.system("curl "+ url + "> ./source/" + url_name + ".html")
print("Done")

trans_2_md.py

import os
import html2text
from lxml import etree

text_maker = html2text.HTML2Text()
text_maker.ignore_links = True
text_maker.bypass_tables = False

files_list = os.listdir('./source')

for filename in files_list:
    with open("./source/" + filename, "r") as f:
        art_html = f.read()
        art_tree = etree.HTML(art_html)
        try:
            art_content = art_tree.xpath("//div[@id='block-aadotcom-content']")[0]
            art_content_html = str(etree.tostring(art_content, method="html"), encoding="utf-8")
            art_content_html = art_content_html.replace("\\n", "")
            art_md = text_maker.handle(str(art_content_html))
            filename = filename.split('.')[0]
            art_md = "## " + filename + "\n" + art_md
            end_num = art_md.rfind('The Scala CookBook')
            art_md = art_md[:end_num]

            with open("./result/" + filename + ".md", "w") as wf:
                wf.write(art_md)
        except:
            print(filename)
print("Done")
